    \frametitle{PV System Desgin and Modeling}
    \only<1>{
        \begin{itemize}
            \item \textbf{Objective:} Maximize overall energy yield of the PV system.
            \item \textbf{Approach:} 
            Group modules by orientation and connect each group to a dedicated inverter with MPPT.\\
            \(\Rightarrow\) Each group operates at its own maximum power point (MPP).
            \item \textbf{Modeling Strategy:} 
            Simulate each group of modules with its corresponding inverter, use the MPP as the operating point, and estimate the total power output as the sum of all group MPPs.
        \end{itemize}
    }
