    \frametitle{Example of a Photovoltaic System}
    \only<1>{
        \small
        \begin{itemize}
            \item consists of 14 identical monocrystalline modules
            \item two strings of seven modules each are connected separately to an
            inverter with two independent MPPT inputs
        \end{itemize}
        \begin{figure}
            \centering
            \includegraphics[scale=0.045]{Surfclub_Dach9.png}
            \caption{\small PV system installed on the roof of a small building.}
        \end{figure}
    }
    \only<2>{
        \small
        \begin{itemize}
            \item \textbf{String:} series connection of several modules
            \item \textbf{Inverter:} converts the direct current (DC) electricity to
            alternating current (AC)
            \item \textbf{MPPT:} Maximum Power Point Tracking dynamically adjusts
            the operating point to its maximum power point (MPP)
        \end{itemize}
